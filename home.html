package com.coverstone.gameobjects;

import com.badlogic.gdx.math.Vector2;
import com.badlogic.gdx.math.Circle;

public class Ship {

private Vector2 position;
private Vector2 velocity;
private Vector2 acceleration;
private int width;
private int height;
private float originalY;

private Circle shipBounds;
private boolean isAlive;

// Constructor for new Ship
public Ship(float x, float y, int width, int height){
	this.width = width;
	this.height = height;
	this.originalY = y;
	position = new Vector2(x, y);
	velocity = new Vector2(0, 0);
	acceleration = new Vector2(0, 900);
	shipBounds = new Circle();
	isAlive = true;
}

// Move ship during gameplay
public void update(float delta){
	velocity.add(acceleration.cpy().scl(delta));

	// Check for max velocity
	if(velocity.y > 470)
	velocity.y = 470;

	// CEILING CHECK
	if(position.y < -15){
	position.y = -15;
	velocity.y = 0;
	}

	position.add(velocity.cpy().scl(delta));
	shipBounds.set(position.x + 17.5f, position.y + 10, 9);
}

// Move ship during Ready state
public void updateReady(float runTime){
	position.x = 50;
	position.y = 6 * (float) Math.sin(12 * runTime) + originalY;
}

// Move ship during Menu State
public void updateMenu(float runTime){
	position.x = (360 / 2 - width - 4);
	position.y = 6 * (float) Math.sin(12 * runTime) + 123;
}

// Set ship back after death
public void onRestart(int y){
	position.y = y;
	velocity.x = 0;
	velocity.y = 0;
	acceleration.x = 0;
	acceleration.y = 900;
	isAlive = true;
}


// Getters, setters and helpers
public boolean addFire(){
	return velocity.y < -200;
}

public void onClick(){
	if(isAlive){
	velocity.y = -329;
	}
}

public float getX(){
	return position.x;
}

public float getY(){
	return position.y;
}

public float getWidth(){
	return width;
}

public float getHeight(){
	return height;
}

public Circle getShipBounds(){
	return shipBounds;
}

public boolean isAlive(){
	return isAlive;
}

public void die(){
	isAlive = false;
}
							
}
